void umlrDriver::scan_begin () {
   yy_flex_debug = trace_scanning;
   if (file.empty () || file == "-") {
       yyin = stdin;
   } else if (!(yyin = fopen (file.c_str (), "r"))) {
              error ("cannot open " + file + ": " + strerror(errno));
              exit (EXIT_FAILURE);
   }
}

void umlrDriver::scan_end () {
  fclose (yyin);
}
